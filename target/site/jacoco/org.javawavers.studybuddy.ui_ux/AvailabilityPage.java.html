  private Integer parseTextFieldValue(TextField textField) {
    if (textField != null) {
      String text = textField.getText();
      if (text == null || text.trim().isEmpty()) {
        return 0;
      } else {
        return Integer.parseInt(text);
      }
    }
    return 0;
  }
